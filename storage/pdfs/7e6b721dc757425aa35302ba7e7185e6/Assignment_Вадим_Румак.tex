\section*{Задание по математике для ученика Вадима}

\begin{enumerate}
    \item Найдите корни уравнения: $3x^2 - 7x + 2 = 0$.
    \item Решите систему уравнений:
    \begin{align*}
        2x + 3y &= 7 \\
        4x - y &= 1
    \end{align*}
    \item Вычислите значение выражения: $\sqrt{5 + 2\sqrt{6}}$.
    \item Найдите сумму бесконечного ряда: $1 + \frac{1}{4} + \frac{1}{9} + \frac{1}{16} + \ldots$.
    \item Разложите на множители: $x^3 - 8$.
\end{enumerate}
