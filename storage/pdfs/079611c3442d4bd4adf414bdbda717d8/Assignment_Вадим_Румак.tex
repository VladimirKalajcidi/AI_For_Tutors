\subsection*{Вопросы повышенной сложности}

\begin{enumerate}
    \item Решите систему уравнений методом подстановки:
    \begin{align*}
    2x + 3y &= 11 \\
    4x - y &= 5
    \end{align*}
    
    \item Найдите сумму бесконечного ряда:
    \[1 + \frac{1}{3} + \frac{1}{9} + \frac{1}{27} + \ldots\]
    
    \item Разложите на множители полином: \(x^3 + 6x^2 + 11x + 6\)
    
    \item Постройте график функции \(y = \frac{x^2 - 4}{x - 2}\)
    
    \item Найдите предел функции \(f(x) = \frac{x^2 - 4}{x - 2}\) при \(x \to 2\)
\end{enumerate}
